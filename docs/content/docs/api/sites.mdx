## Update Site Config

<div className="grid grid-cols-1 xl:grid-cols-2 gap-8">
<div>

```
PUT /api/sites/:siteId/config
```

Updates site configuration settings. All fields are optional - only include the fields you want to update. Requires admin or owner role.

### Path Parameters

<TypeTable
  type={{
    siteId: {
      description: 'Site ID',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Request Body

<TypeTable
  type={{
    domain: {
      description: 'Site domain',
      type: 'string',
      typeDescription: 'Valid domain format (e.g., example.com)',
      required: false,
    },
    public: {
      description: 'Make analytics publicly accessible',
      type: 'boolean',
      required: false,
    },
    saltUserIds: {
      description: 'Salt user IDs for privacy',
      type: 'boolean',
      required: false,
    },
    blockBots: {
      description: 'Block bot traffic',
      type: 'boolean',
      required: false,
    },
    excludedIPs: {
      description: 'List of IP addresses or CIDR ranges to exclude',
      type: 'string[]',
      typeDescription: 'Max 100 entries',
      required: false,
    },
    excludedCountries: {
      description: 'List of country codes to exclude',
      type: 'string[]',
      typeDescription: '2-letter ISO codes (e.g., US, GB)',
      required: false,
    },
    sessionReplay: {
      description: 'Enable session replay',
      type: 'boolean',
      required: false,
    },
    webVitals: {
      description: 'Enable Web Vitals tracking',
      type: 'boolean',
      required: false,
    },
    trackErrors: {
      description: 'Enable error tracking',
      type: 'boolean',
      required: false,
    },
    trackOutbound: {
      description: 'Enable outbound link tracking',
      type: 'boolean',
      required: false,
    },
    trackUrlParams: {
      description: 'Enable URL parameter tracking',
      type: 'boolean',
      required: false,
    },
  }}
/>

### Response

<TypeTable
  type={{
    success: {
      description: 'Whether the update was successful',
      type: 'boolean',
    },
    message: {
      description: 'Status message',
      type: 'string',
    },
    config: {
      description: 'Updated site configuration',
      type: 'object',
    },
  }}
/>

</div>
